 63 /**
 64  * class for TSP TimeStampToken ASN.1 object<br/>
 65  * @name KJUR.asn1.tsp.TimeStampToken
 66  * @class class for TSP TimeStampToken ASN.1 object
 67  * @param {Array} params JSON object for constructor parameters
 68  * @extends KJUR.asn1.cms.SignedData
 69  * @since jsrsasign 10.0.0 asn1tsp 2.0.0
 70  * @see KJUR.asn1.tsp.TimeStampResp
 71  * @see KJUR.asn1.tsp.TSTInfo
 72  *
 73  * @description
 74  * This is an ASN.1 encoder for TimeStampToken
 75  * ASN.1 structure defined in
 76  * <a href="https://tools.ietf.org/html/rfc3161#section-2.4.2">
 77  * RFC 3161 TSP section 2.4.2</a>.
 78  * <pre>
 79  * TimeStampToken ::= ContentInfo
 80  *   -- contentType is id-signedData ([CMS])
 81  *   -- content is SignedData ([CMS])
 82  * id-ct-TSTInfo  OBJECT IDENTIFIER ::= { iso(1) member-body(2)
 83  * us(840) rsadsi(113549) pkcs(1) pkcs-9(9) smime(16) ct(1) 4}
 84  * </pre>
 85  * Constructor argument "params" is similar to
 86  * {@link KJUR.asn1.cms.SignedData} however "econtent"
 87  * value is different as follows:
 88  * <ul>
 89  * <li>econtent.type - shall be "tstinfo"</li>
 90  * <li>econtent.content - shall be {@link KJUR.asn1.tsp.TSTInfo} parameter</li>
 91  * </ul>
 92  *
 93  * @example
 94  * new KJUR.asn1.tsp.TimeStampToken({
 95  *   version: 1,
 96  *   hashalgs: ["sha256"],
 97  *   econtent: {
 98  *     type: "tstinfo",
 99  *     content: {
100  *       policy: '1.2.3.4.5',
101  *       messageImprint: { hashAlg: 'sha1', hashValue: 'a1a2a3a4' },
102  *       serialNumber: {'int': 3},
103  *       genTime: {str: '20131231235959.123Z', millis: true},
104  *       accuracy: { millis: 500 },
105  *       ordering: true,
106  *       nonce: {'int': 3},
107  *     }
108  *   },
109  *   sinfos: [{
110  *     version: 1,
111  *     id: {type:'isssn', cert: sZ4_CERPEM},
112  *     hashalg: "sha256",
113  *     sattrs: {array: [{
114  *       attr: "contentType",
115  *       type: "data"
116  *     },{
117  *       attr: "signingTime",
118  *       str: '131231235959Z'
119  *     },{
120  *       attr: "messageDigest",
121  *       hex: 'ffff'
122  *     }]},
123  *     sigalg: "SHA256withRSA",
124  *     signkey: sZ4_PRVP8PPEM
125  *   }]
126  * })
127  */
128 KJUR.asn1.tsp.TimeStampToken = function(params) {
129     var _KJUR = KJUR,
130 	_KJUR_asn1 = _KJUR.asn1,
131 	_KJUR_asn1_tsp = _KJUR_asn1.tsp;
132 
133     _KJUR_asn1_tsp.TimeStampToken.superclass.constructor.call(this);
134 
135     this.params = null;
136 
137     this.getEncodedHexPrepare = function() {
138 	//alert("getEncodedHexPrepare called...");
139 	var dTSTInfo = new _KJUR_asn1_tsp.TSTInfo(this.params.econtent.content);
140 	this.params.econtent.content.hex = dTSTInfo.getEncodedHex();
141     };
142 
143     if (params != undefined) this.setByParam(params);
144 };
145 YAHOO.lang.extend(KJUR.asn1.tsp.TimeStampToken, KJUR.asn1.cms.SignedData);

147 /**
148  * class for TSP TSTInfo ASN.1 object
149  * @name KJUR.asn1.tsp.TSTInfo
150  * @class class for TSP TSTInfo ASN.1 object
151  * @param {Array} params JSON object for TSTInfo parameters
152  * @extends KJUR.asn1.ASN1Object
153  * @since jsrsasign 4.6.0 asn1tsp 1.0.0
154  * @see KJUR.asn1.x509.X500Name
155  * @see KJUR.asn1.x509.GeneralName
156  * @description
157  * This class represents TSTInfo ASN.1 structure.
158  * <pre>
159  * TSTInfo ::= SEQUENCE  {
160  *    version         INTEGER  { v1(1) },
161  *    policy          TSAPolicyId,
162  *    messageImprint  MessageImprint,
163  *    serialNumber    INTEGER, -- up to 160bit
164  *    genTime         GeneralizedTime,
165  *    accuracy        Accuracy                 OPTIONAL,
166  *    ordering        BOOLEAN                  DEFAULT FALSE,
167  *    nonce           INTEGER                  OPTIONAL,
168  *    tsa             [0] GeneralName          OPTIONAL,
169  *    extensions      [1] IMPLICIT Extensions  OPTIONAL   }
170  * </pre>
171  * For "params" arguent, following properties are accepted:
172  * <ul>
173  * <li>{Array}tsa - {@link KJUR.asn1.x509.X500Name} parameter for
174  * tsa field even though tsa field is GeneralName.</li>
175  * </ul>
176  * @example
177  * o = new KJUR.asn1.tsp.TSTInfo({
178  *     policy:    '1.2.3.4.5',
179  *     messageImprint: {alg: 'sha256', hash: '1abc...'},
180  *     genTime:   {millis: true},         // OPTION
181  *     accuracy:  {micros: 500},          // OPTION
182  *     ordering:  true,                   // OPITON
183  *     nonce:     {hex: '52fab1...'},     // OPTION
184  *     tsa:       {str: '/C=US/O=TSA1'}   // OPITON
185  * });
186  */
187 KJUR.asn1.tsp.TSTInfo = function(params) {
188     var _KJUR = KJUR,
189 	_KJUR_asn1 = _KJUR.asn1,
190 	_DERSequence = _KJUR_asn1.DERSequence,
191 	_DERInteger = _KJUR_asn1.DERInteger,
192 	_DERBoolean = _KJUR_asn1.DERBoolean,
193 	_DERGeneralizedTime = _KJUR_asn1.DERGeneralizedTime,
194 	_DERObjectIdentifier = _KJUR_asn1.DERObjectIdentifier,
195 	_DERTaggedObject = _KJUR_asn1.DERTaggedObject,
196 	_KJUR_asn1_tsp = _KJUR_asn1.tsp,
197 	_MessageImprint = _KJUR_asn1_tsp.MessageImprint,
198 	_Accuracy = _KJUR_asn1_tsp.Accuracy,
199         _X500Name = _KJUR_asn1.x509.X500Name,
200         _GeneralName = _KJUR_asn1.x509.GeneralName;
201 	
202 
203     _KJUR_asn1_tsp.TSTInfo.superclass.constructor.call(this);
204 
205     this.dVersion = new _DERInteger({'int': 1});
206     this.dPolicy = null;
207     this.dMessageImprint = null;
208     this.dSerialNumber = null;
209     this.dGenTime = null;
210     this.dAccuracy = null;
211     this.dOrdering = null;
212     this.dNonce = null;
213     this.dTsa = null;
214 
215     this.getEncodedHex = function() {
216         var a = [this.dVersion];
217 
218         if (this.dPolicy == null) throw "policy shall be specified.";
219         a.push(this.dPolicy);
220 
221         if (this.dMessageImprint == null)
222             throw "messageImprint shall be specified.";
223         a.push(this.dMessageImprint);
224 
225         if (this.dSerialNumber == null)
226             throw "serialNumber shall be specified.";
227         a.push(this.dSerialNumber);
228 
229         if (this.dGenTime == null)
230             throw "genTime shall be specified.";
231         a.push(this.dGenTime);
232 
233         if (this.dAccuracy != null) a.push(this.dAccuracy);
234         if (this.dOrdering != null) a.push(this.dOrdering);
235         if (this.dNonce != null) a.push(this.dNonce);
236         if (this.dTsa != null) a.push(this.dTsa);
237 
238         var seq = new _DERSequence({array: a});
239         this.hTLV = seq.getEncodedHex();
240         return this.hTLV;
241     };
242 
243     if (params !== undefined) {
244         if (typeof params.policy == "string") {
245             if (! params.policy.match(/^[0-9.]+$/))
246                 throw "policy shall be oid like 0.1.4.134";
247             this.dPolicy = new _DERObjectIdentifier({oid: params.policy});
248         }
249         if (params.messageImprint !== undefined) {
250             this.dMessageImprint = new _MessageImprint(params.messageImprint);
251         }
252         if (params.serialNumber !== undefined) {
253             this.dSerialNumber = new _DERInteger(params.serialNumber);
254         }
255         if (params.genTime !== undefined) {
256             this.dGenTime = new _DERGeneralizedTime(params.genTime);
257         }
258         if (params.accuracy !== undefined) {
259             this.dAccuracy = new _Accuracy(params.accuracy);
260         }
261         if (params.ordering !== undefined &&
262             params.ordering == true) {
263             this.dOrdering = new _DERBoolean();
264         }
265         if (params.nonce !== undefined) {
266             this.dNonce = new _DERInteger(params.nonce);
267         }
268         if (params.tsa !== undefined) {
269             this.dTsa = new _DERTaggedObject({
270 		tag: "a0",
271 		explicit: true,
272 		obj: new _GeneralName({dn: params.tsa})
273 	    });
274         }
275     }
276 };
277 YAHOO.lang.extend(KJUR.asn1.tsp.TSTInfo, KJUR.asn1.ASN1Object);

279 /**
280  * class for TSP Accuracy ASN.1 object
281  * @name KJUR.asn1.tsp.Accuracy
282  * @class class for TSP Accuracy ASN.1 object
283  * @param {Array} params associative array of parameters
284  * @extends KJUR.asn1.ASN1Object
285  * @since jsrsasign 4.6.0 asn1tsp 1.0.0
286  *
287  * @description
288  * This is an ASN.1 encoder for Accuracy
289  * ASN.1 structure defined in
290  * <a href="https://tools.ietf.org/html/rfc3161#section-2.4.2">
291  * RFC 3161 TSP section 2.4.2</a>.
292  * <pre>
293  * Accuracy ::= SEQUENCE {
294  *    seconds        INTEGER              OPTIONAL,
295  *    millis     [0] INTEGER  (1..999)    OPTIONAL,
296  *    micros     [1] INTEGER  (1..999)    OPTIONAL  }
297  * </pre>
298  *
299  * @example
300  * new KJUR.asn1.tsp.Accuracy({
301  *   seconds: 1,   // OPTION
302  *   millis: 500,  // OPTION
303  *   micros: 500   // OPTION
304  * });
305  */
306 KJUR.asn1.tsp.Accuracy = function(params) {
307     var _KJUR = KJUR,
308 	_KJUR_asn1 = _KJUR.asn1,
309 	_newObject = _KJUR_asn1.ASN1Util.newObject;
310 
311     _KJUR_asn1.tsp.Accuracy.superclass.constructor.call(this);
312 
313     this.params = null;
314 
315     this.getEncodedHex = function() {
316 	var params = this.params;
317 	var a = [];
318 	if (params.seconds != undefined &&
319 	    typeof params.seconds == "number") {
320 	    a.push({"int": params.seconds});
321 	}
322 	if (params.millis != undefined &&
323 	    typeof params.millis == "number") {
324 	    a.push({tag: {tagi:"80", obj:{"int": params.millis}}});
325 	}
326 	if (params.micros != undefined &&
327 	    typeof params.micros == "number") {
328 	    a.push({tag: {tagi:"81", obj:{"int": params.micros}}});
329 	}
330 	return _newObject({"seq": a}).getEncodedHex();
331     };
332 
333     if (params != undefined) this.setByParam(params);
334 };

364  *   alg: 'sha256',
365  *   hash: '1f3dea...'
366  * });
367  */
368 KJUR.asn1.tsp.MessageImprint = function(params) {
369     var _KJUR = KJUR,

371 	_DERSequence = _KJUR_asn1.DERSequence,

510 /**
511  * class for TSP PKIStatusInfo ASN.1 object
512  * @name KJUR.asn1.tsp.PKIStatusInfo
513  * @class class for TSP PKIStatusInfo ASN.1 object
514  * @param {Array} params associative array of parameters
515  * @extends KJUR.asn1.ASN1Object
516  * @since jsrsasign 4.6.0 asn1tsp 1.0.0
517  * @description
518  * <pre>
519  * PKIStatusInfo ::= SEQUENCE {
520  *    status                  PKIStatus,
521  *    statusString            PKIFreeText     OPTIONAL,
522  *    failInfo                PKIFailureInfo  OPTIONAL  }
523  * </pre>
524  *
525  * @example
526  * new KJUR.asn1.tsp.PKIStatusInfo("granted")
527  * new KJUR.asn1.tsp.PKIStatusInfo({status: "granted"})
528  * new KJUR.asn1.tsp.PKIStatusInfo({
529  *   status: 2, // rejection
530  *   statusString: ["unsupported algorithm"], // OPTION
531  *   failInfo: 'badAlg' // OPTION
532  * })
533  */
534 KJUR.asn1.tsp.PKIStatusInfo = function(params) {
535     var _Error = Error,
536 	_KJUR = KJUR,
537 	_KJUR_asn1 = _KJUR.asn1,
538 	_DERSequence = _KJUR_asn1.DERSequence,
539 	_KJUR_asn1_tsp = _KJUR_asn1.tsp,
540 	_PKIStatus = _KJUR_asn1_tsp.PKIStatus,
541 	_PKIFreeText = _KJUR_asn1_tsp.PKIFreeText,
542 	_PKIFailureInfo = _KJUR_asn1_tsp.PKIFailureInfo;
543 
544     _KJUR_asn1_tsp.PKIStatusInfo.superclass.constructor.call(this);
545 
546     this.params = null;
547 
548     this.getEncodedHex = function() {
549 	var params = this.params;
550 
551 	var a = [];

916  *  nonce: '9abcf318...',            // nonce (OPTION)
